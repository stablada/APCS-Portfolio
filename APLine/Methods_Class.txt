public class Methods { //Methods class does all the calculations for the whole program and is called in the game class.
  public static double slope(int a, int b) { // returns the slope
    double avg = ((0 - a) / b);
    return avg;
  }
  
  public static void pointLoc(int a, int b, int c) { //tries to test if a point is on the specified line
    int x = 0;
    int y = 0;
    int calc = 0;
    x = Integer.parseInt(JOP.in("Plug in the X coordinate."));
    y = Integer.parseInt(JOP.in("Plug in the Y coordinate."));
    calc = a * x;
    calc += b * y;
    calc += c;
    if (calc == 0) {
      JOP.msg("This point is on the line!");
      Game.display();
    } else if (calc > 0) {
      JOP.msg("This point is above the line!");
      Game.display();
    } else if (calc < 0) {
      JOP.msg("This point is below the line!");
      Game.display();
    } 
  }
  
  public static void addSlope() { //adds slope to the line
    Game.a *= Integer.parseInt(JOP.in("What would you like the slope to be multiplied by?"));
  }
  
  public static void addInter() { //adds the y intercept to the line
    Game.yint = Integer.parseInt(JOP.in("What would you like the y intercept to be?"));
  }
  
  public static void interLoc(int a, int b, int c) { //tries to find intersection between two lines and the point at which they meet
    int x = 0;
    int y = 0;
    int z = 0;
    int calc = 0;
    x = Integer.parseInt(JOP.in("Plug in the a value."));
    y = Integer.parseInt(JOP.in("Plug in the b value."));
    z = Integer.parseInt(JOP.in("Plug in the c value."));
    calc = a * x;
    calc += b * y;
    calc += c + z;
    if (calc > 0) {
      JOP.msg("This line intersects!");
      Game.display();
    } else if (calc == 0) {
      JOP.msg("This line does not intersect\nIt is below the other line");
      Game.display();
    } else if (calc < 0) {
      JOP.msg("This line does not intersect\nIt is above the other line");
      Game.display();
    } 
  }
}
