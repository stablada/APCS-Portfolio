package Model;

import View.JOP;

public class Game { \\Game Class
  private static String gameMode = " ";
  
  public static void Startup() { \\Startup Method - Begins game and starts all popups for UI Direction to PvP or PvCPU
    boolean runQuery = true;
    JOP.msg("Welcome to Rock Paper Scissors! Please Select a game mode to begin!");
    while (runQuery) {
      gameMode = JOP.in("Please type [CPU] to play against a computer or [PvP] to play with a friend");
      if (gameMode.equalsIgnoreCase("CPU")) {
        runQuery = false;
        cpuHelpDialog();
        CPU.PvC();
        continue;
      } 
      if (gameMode.equalsIgnoreCase("PVP")) {
        runQuery = false;
        playerHelpDialog();
        Player.PvP();
      } 
    } 
  }
  
  public static void playerHelpDialog() { \\Shows JOP for PvP help menu
    JOP.msg("Game Rules:\nThis game tracks wins for each player up to 3 rounds, then gives one point to the total winner.\nTo play:\nPlayers take turns typing in their choice, without the other looking at the screen.\nTip: Make sure to designate your status as player one or player two to the other player and keep track of your turns.\nCommands:\n[Exit] - Exits the program entirely.\n[Help] - Reopens this screen.\n[Reset] - Resets the whole program, starting from the beginning and clearing all scores.");
  }
  
  public static void cpuHelpDialog() { \\Shows JOP for PvCPU help menu
    JOP.msg("Game Rules:\nThis game tracks your wins against the Computer for up to 3 rounds, then gives one point to the total winner.\nTo play:\nType your choice into the type box to make a move.\nThe result of the game will be displayed immidiately after your move.\nCommands:\n[Exit] - Exits the program entirely.\n[Help] - Reopens this screen.\n[Reset] - Resets the whole program, starting from the beginning and clearing all scores.");
  }
}
